export const HowToDownload = ({ resource, requirements }) => {
    return (
        <>
            <div>
                <h2>Requirements</h2>

                {requirements.length === 0 ? (
                    <Note>สคริปต์นี้ไม่มี Requirements</Note>
                ) : (
                    <ul>
                        {
                            requirements.map((data, index) => (
                                <li>
                                    {data.name} ({data.version})           
                                </li>
                            ))
                        }
                    </ul>
                )}
            </div>

            <Steps>
                <Step title="Download Resource">
                    ให้คุณทำการดาวน์โหลดไฟล์ Script จาก <a href="https://msc-fivem.shop" target="_blank">เว็บไซต์</a> ของเราหลังจากที่ได้ชำระเงินแล้วซื้อสินค้าเสร็จสิ้น
                </Step>

                <Step title="Move To Your Server Resources Folder">
                    ให้ทำการแตกไฟล์แล้วนำโฟลเดอร์ <code>{resource}</code> ไปวางไว้ภายในโฟลเดอร์ <code>resources/{resource}</code>
                </Step>

                <Step title="Add Resource To Start When Server Is Starting Up">
                    เมื่อคุณนำไฟล์ลงที่เซิฟเวอร์สำเร็จให้ไปที่ไฟล์ <code>server.cfg</code> และเพิ่ม Code ดังนี้

                    <CodeBlock language="lua" filename="server.cfg">
                        {`ensure ${resource}`}
                    </CodeBlock>
                </Step>
            </Steps>

            <Note>เมื่อเซิฟเวอร์ทำงานแล้วให้ตรวจสอบ Server Console ว่าสคริปต์ได้ทำงานหรือยัง และตรวจสอบ License ถูกต้องหรือไม่</Note>
        </>
    )
}
